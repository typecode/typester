Module: core/Context

Context - Instance safe context builder that can mixin multiple additional objects as contexts. These can then be used to bind methods into a shared context.
Source:
  • core/Context.js, line 4
Example
let context = new Context({ ping: 'pong' });
// context.ping = 'pong'

context.mixin({ foo: 'bar' }, { jim: 'jam' })
// context.foo = 'bar'
// context.jim = 'jam'

context.extendWith({ bing: 'bong', bang: 'boom' }, { keys: ['bing'] })
// context.bing = 'bong'
// context.bang = undefined
